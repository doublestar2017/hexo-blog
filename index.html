中文文档

在vue中使用

安装
npm install --save three

引入

import * as THREE from 'three'
import { GLTFLoader } from 'three/examples/jsm/loaders/GLTFLoader.js'
// 轨道控制器
import { OrbitControls } from 'three/examples/jsm/controls/OrbitControls'

创建与加载模型

// 创建场景
  const scene = new THREE.Scene()
  // 灰shai背景
  scene.background = new THREE.Color("#eee")
  // 雾
  scene.fog = new THREE.Fog('#eee', 20, 100)

  // 设置纹理，贴图
  const canvas = document.querySelector('#three')
  const renderer = new THREE.WebGLRenderer({ canvas, antialias: true })
  // 影子
  renderer.shadowMap.enabled = true

  // PerspectiveCamera（透视摄像机）以及OrthographicCamera （正交投影摄像机）
  // 创建摄像机
  // PerspectiveCamera( fov : Number, aspect : Number, near : Number, far : Number )具有四个参数:
  const camera = new THREE.PerspectiveCamera(50, window.innerWidth / window.innerHeight, 0.1, 1000)
  camera.position.z = 10

  // 创建地板
  let floorGeometry = new THREE.PlaneGeometry(3000, 3000)
  let floorMaterial = new THREE.MeshPhongMaterial({ color: 0x000000 })
  let floor = new THREE.Mesh(floorGeometry, floorMaterial)
  floor.rotation.x = -0.5 * Math.PI
  floor.receiveShadow = true
  floor.position.y = -0.001
  scene.add(floor)

  // 添加个平行光
  const dirLight = new THREE.DirectionalLight(0xffffff, 0.6)
  // 光源位置
  dirLight.position.set(-10, 8, -5)
  // 产生阴影
  dirLight.castShadow = true
  dirLight.shadow.mapSize = new THREE.Vector2(1024, 1024)
  scene.add(dirLight)

  // 添加多个光源
  const hemLight = new THREE.HemisphereLight(0xffffff, 0xffffff, 0.6)
  hemLight.position.set(0, 48, 0)
  scene.add(hemLight)

  const controls = new OrbitControls(camera, renderer.domElement)
  // 加点阻尼感
  controls.enableDamping = true

  function animate () {
    controls.update()

    renderer.render(scene, camera)
    requestAnimationFrame(animate)

    // 分辨率问题
    if (resizeRendererToDisplaySize(renderer)) {
      const canvas = renderer.domElement
      camera.aspect = canvas.clientWidth / canvas.clientHeight
      camera.updateProjectionMatrix()
    }
  }

  const gltfLoader = new GLTFLoader()
  gltfLoader.load('/tyrannosaurus_rex_skeleton/scene.gltf', (gltf) => {
    var model = gltf.scene
    // 添加这段代码
    // 遍历模型每部分
    model.traverse((o) => {
      // 将图片作为纹理加载
      let explosionText = new THREE.TextureLoader().load('/tyrannosaurus_rex_skeleton/textures/museumhallRDbone_baseColor.jpeg')
      explosionText.flipY = false
      // 将纹理图生成基础网络材质（MeshBasicMaterial）
      const material = new THREE.MeshBasicMaterial({
        map: explosionText
      })
      // 给模型每个部分材质
      o.material = material

      // 模型每个部分都能产生阴影
      if (o.isMesh) {
        o.castShadow = true
        o.receiveShadow = true
      }
    })
    scene.add(model)
  })

  // devicePixelRatio此属性返回当前显示的设备的物理像素分辨率与CSS像素分辨率的比值。
  // 该值也可以被解释为像素大小的比例，即一个CSS像素的大小相对于一个物理像素的大小的比值
  function resizeRendererToDisplaySize (renderer) {
    const canvas = renderer.domElement
    var width = window.innerWidth
    var height = window.innerHeight
    var canvasPixelWidth = canvas.width / window.devicePixelRatio
    var canvasPixelHeight = canvas.height / window.devicePixelRatio

    const needResize = canvasPixelWidth !== width || canvasPixelHeight !== height
    if (needResize) {
      renderer.setSize(width, height, false)
      return needResize
    }

  }
